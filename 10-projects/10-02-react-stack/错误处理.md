export function ComponentErrorBoundary({
  children,
  fallback = ComponentError,
  resetKeys = [],
}: ComponentErrorBoundaryProps) {
  const handleError = useErrorHandler();

  handleError(new Error(message));
  return (
    <ErrorBoundary
      FallbackComponent={fallback}

      resetKeys={resetKeys}
      isolate   
    >
      {children}
    </ErrorBoundary>
  );
}
